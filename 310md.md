using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading;

namespace CSD03
{
    public abstract class LottoryBase : IDisposable
    {
        private List<IEnumerable<int>> _lottoryNumberContext = null;
        private Random _rnd = null;

        public LottoryBase()
        {
            this._lottoryNumberContext = new List<IEnumerable<int>>(    );
            this._rnd = new Random();
        }

        protected IEnumerable<int> GetLottoryItem()
        {
            if (this._lottoryNumberContext.Count == 0)
                return null;
            
            int index = _rnd.Next(0, this._lottoryNumberContext.Count - 1);
            return this._lottoryNumberContext[index];
        }

        protected void CreateLottoryNumberSequence(int Min, int Max, int Length)
        {
            if (this._lottoryNumberContext.Count != 0)
                return;

            for (int i = 0; i < 100; i++)
            {
                this.CreateLottoryNumberSequenceInternal(Min, Max, Length); 
                Thread.Sleep(5);
            }
        }

        private void CreateLottoryNumberSequenceInternal(int Min, int Max, int Length)
        {
            List<int> numbers = new List<int>();
            Random rnd = new Random();

            while (numbers.Count < Length)
            {
                int number = 0;

                do
                {
                    number = rnd.Next(Min, Max);
                }
                while (numbers.Contains(number));

                numbers.Add(number);
            }

            this._lottoryNumberContext.Add(numbers);
        }

        public abstract IEnumerable<int> GetLottorySequence();

        void IDisposable.Dispose()
        {
            this._lottoryNumberContext.Clear();
            this._lottoryNumberContext = null;
            GC.SuppressFinalize(this);
        }
    }
}
